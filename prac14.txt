create table emp2 with fields empno as int, name as char, category as int and write a trigger for insert or update on emp2 for new record in uppercase
only
using command like name= upper(name)

create table emp2 (
	empno int;
	name varchar2(10);
	category int);

create or replace trigger emp2_undercase2
before insert or update on emp2
for each row
begin
:new.name := upper(:new.name);
end;
/

trigger for salary updation:

create or replace trigger salaryupdate
before insert or update on emp2
for each row
begin
if :new.salary < 10000 and :new.category = 1 then
	dbms_output.put_line('Salary should be more than 10000');
elsif :new.salary < 20000 and :new.category = 2 then
	dbms_output.put_line('Salary should be more than 20000');
elsif :new.salary < 30000 and :new.category = 3 then
	dbms_output.put_line('Salary should be more than 30000');
end if;
end;
/
