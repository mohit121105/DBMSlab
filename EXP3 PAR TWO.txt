2 RETRIEVE DISTINCTNUMBER OF INSTRUCTOR FROM EACH DEPARTMENT
select count(distinct i.name) as total_names_per_dept from instructor i join department d on i.dept_id = d.dept_id group by d.name;

3. FIND THE NAMES OF THE INSTRUCTORS EARNING MORE THAN BIOLOFY INSTRUCTOR
SELECT i.name,d.name,i.salary FROM instructor i JOIN department d ON i.dept_id=d.dept_id WHERE i.salary>ANY(SELECT i2.salary FROM instructor i2 JOIN department d2 ON i2.dept_id=d2.dept_id WHERE d2.name='Biology');
