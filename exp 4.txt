1 display all employees name start with 'a' and are at least 3 char in length

select first_name form emp where first_name like 'a%' and length(first_name) >= 3;

2 display departments having Dallas, New York and Chicago locations

select DEPT_NAME from departments where lower(location) like 'da%' or lower(location) like 'ne%' or lower(location) like 'ch%' 

3 display employees who are clerk , salesman , analyst earning more than 3000
  
select FIRST_NAME, job, salary from emp where (lower(job) = 'clerk' or lower(job) = 'clerk' or lower(job) = 'salesman' or lower(job) = 'analyst') and salary > 3000;

4 display the name of emloyees who are not working as managers
select FIRST_NAME, job from emp where (lower(job) != 'manager');
